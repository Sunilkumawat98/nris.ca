curl --request GET \
    --get "https://stage-api.nris.ca/api/v1/get-all-category" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"*user_id\": 1,
    \"*state_id\": 1,
    \"*title\": \"\'This is test title\' in JSON BODY\",
    \"*description\": \"\'This is test description\' in JSON BODY\"
}"

const url = new URL(
    "https://stage-api.nris.ca/api/v1/get-all-category"
